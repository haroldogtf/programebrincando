APRENDA triangulo (tamanho)
	CORCANETA COR(165,42,42);
	GIRARESQUERDA 60; PARAFRENTE tamanho;
	GIRARDIREITA 120; PARAFRENTE tamanho;
	GIRARDIREITA 120; PARAFRENTE tamanho
FIMAPRENDA;

APRENDA quadrado (tamanho)
	VAR i = 4;
	
	REPETIR i > 0,
		PARAFRENTE tamanho;
		GIRARDIREITA 90;
		i = i - 1 
	FIMREPETIR
FIMAPRENDA;

CRIA casa = quadrado JUNTA triangulo;

DESENHE casa(100, 100)
